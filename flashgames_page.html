import os import re import json # ================================================================ # 🔧 ခင်ဗျား Internet Archive Item ID ကို ဒီမှာ ထည့်ပါ # ================================================================ IA_ITEM_ID = "my_flash_games_arcade" # 👈 ခင်ဗျား သတ်မှတ်ထားတဲ့ ID နဲ့ အစားထိုးပါ BASE_IA_URL = f"https://archive.org/download/{IA_ITEM_ID}/" # ================================================================ def update_game_list(): games_folder = "games" html_file = "flashgames_page.html" if not os.path.exists(games_folder): print("❌ 'games' folder မတွေ့ပါ။") return swf_files = [] for f in os.listdir(games_folder): if f.lower().endswith('.swf'): raw_name = os.path.splitext(f)[0] display_name = raw_name.replace('_', ' ').replace('-', ' ').title() # 🔍 တူညီတဲ့နာမည်ရှိ ပုံဖိုင်ကို ရှာမယ် thumb_url = "" for ext in ['.jpg', '.png', '.jpeg', '.gif', '.webp']: img_path = os.path.join(games_folder, raw_name + ext) if os.path.exists(img_path): # Internet Archive ပေါ်က ပုံလိပ်စာ thumb_url = BASE_IA_URL + raw_name + ext break swf_files.append({ "name": display_name, "file": BASE_IA_URL + f, # SWF ဖိုင် ဒေါင်းလုဒ်လိပ်စာ "thumb": thumb_url, # 🖼️ Preview ပုံလိပ်စာ "emoji": "🎮" }) if not swf_files: print("⚠️ 'games' folder ထဲမှာ .swf ဖိုင်တွေ မတွေ့ပါ။") return # JavaScript Array ကို ဆောက်မယ် js_array = "const games = " + json.dumps(swf_files, indent=4, ensure_ascii=False) + ";" # HTML ထဲက ဟောင်းနေတဲ့ Array ကို အစားထိုးမယ် if not os.path.exists(html_file): print(f"❌ '{html_file}' ဖိုင် မတွေ့ပါ။") return with open(html_file, 'r', encoding='utf-8') as f: content = f.read() pattern = r'const games = \[.*?\];' new_content = re.sub(pattern, lambda m: js_array, content, flags=re.DOTALL) if new_content == content: print("⚠️ 'const games = [...]' ကို HTML ထဲမှာ မတွေ့ပါ။") return with open(html_file, 'w', encoding='utf-8') as f: f.write(new_content) print(f"✅ အောင်မြင်ပါပြီ။ {len(swf_files)} ခု ဂိမ်းတွေကို {html_file} ထဲ ထည့်လိုက်ပါပြီ။") if __name__ == "__main__": update_game_list()
